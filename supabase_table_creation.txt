//table - pedidos

create table
  public.pedidos (
    id bigint generated by default as identity,
    created_at timestamp with time zone not null default now(),
    total numeric null,
    constraint pedidos_pkey primary key (id)
  ) tablespace pg_default;


//table - avisos

create table
  public.avisos (
    id bigint generated by default as identity,
    created_at timestamp with time zone not null default now(),
    vigencia date null,
    aviso text null,
    url_image text null,
    constraint avisos_pkey primary key (id)
  ) tablespace pg_default;


//table - produto

create table
  public.produto (
    id bigint generated by default as identity,
    created_at timestamp with time zone not null default now(),
    descrição text null,
    valor numeric null,
    extensao_file text null,
    categoria text null,
    nome text null,
    url_image text null,
    constraint produto_pkey primary key (id)
  ) tablespace pg_default;
